Realizar un programa el cual solicite el nombre de su mascota, edad de la mascota, el tipo de mascota y su nombre completo.                 package variables.objects;

                public class mascota {
                    String nombre;
                    String tipo;
                    String edad;
                    String dueno;
                
                    Scanner entrada = new Scanner(System.in);
                
                    public void impresion() {
                
                        System.out.println("Cual es el nombre de la mascota");
                        nombre = entrada.nextLine();
                        System.out.println("Cual es el tipo de mascota");
                        tipo = entrada.nextLine();
                        System.out.println("Cual es la edad de la mascota");
                        edad = entrada.nextLine();

                        dueno = entrada.nextLine();
                
                        System.out.println(nombre + " es un(a) " + tipo + ", el cual, tiene " + edad + " años de edad y " + dueno + " es actualmente su dueño(a).");
                
                    }
                }
