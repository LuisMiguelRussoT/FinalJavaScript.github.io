                public class Matriz {
                    int fila;

                    public int[][] matrizNormal() {
                        int[][] normal;
                        normal = new int[4][5];
                        int contador = 1;
                
                        for (fila = 0; fila <= 3; fila++) {
                            for (columna = 0; columna <= 4; columna++) {
                                normal[fila][columna] = contador;
                                contador = contador + 1;
                
                                if (contador <= 10) {
                                    System.out.print("0" + normal[fila][columna] + " ");
                                }
                                if (contador > 10) {
                                    System.out.print(normal[fila][columna] + " ");
                                }
                            }
                            System.out.println("");
                        }
                        return normal;
                    }
                
                
                    public void invertida(int[][] matriz) {
                
                        int contador = 0;
                        fila = 0;
                        columna = 0;
                        while (contador <= 3) {
                
                            if (fila == 0 || fila == 2) {
                
                                for (columna = 0; columna <= 4; columna++) {
                
                                    if (matriz[fila][columna] < 10) {
                                        System.out.print("0" + matriz[fila][columna] + " ");
                                    }
                                    if (matriz[fila][columna] >= 10) {
                                        System.out.print(matriz[fila][columna] + " ");
                                    }
                
                                }
                                System.out.println("");
                            }
                
                            if(fila == 1 || fila == 3){
                                for (columna = 4; columna >= 0; columna--) {
                
                                    if (matriz[fila][columna] < 10) {
                                        System.out.print("0" + matriz[fila][columna] + " ");
                                    }
                                    if (matriz[fila][columna] >= 10) {
                                        System.out.print(matriz[fila][columna] + " ");
                                    }
                                }
                                System.out.println("");
                
                            }
                            contador = contador + 1;
                            fila = fila + 1;
                        }
                    }
                }
